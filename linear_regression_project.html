def transpose(M):
    # Mrow, col
    row, col = shape(M)
    
    # Trans Matrix[row, col], [col, row]
    # trans matrix
    trans = [ [ 0 for c in range(row) ] for r in range(col) ]
    
    # M, trans matrix
    r = 0
    while r < row:
        c = 0
        while c < col:
            trans[c][r] = M[r][c] 
            c += 1
        r += 1
        
    # Trans Matrix
    return(trans)

    # A,Brow, col
    arow, acol = shape(A)
    brow, bcol = shape(B)
    p = acol
    
    # , A == B,

    result = []
    for x in range(arow):
        # result[row]
        row_list = []
        # Bcol
        for y in range(bcol):
            # collist
            col_list = get_col(B, y)
            # A[x]B[n][y], n

            row_list.append(rs)
        # row_listresult[row]
        result.append(row_list)
    # result
    return(result)

def trans_row(M, r, v):
    
    # 
    row, col = shape(M)
    
    # M[n][v], n, , M[r]
    x = r + 1
    while x < row:
        if not is_zero(M[x][v]):
            swapRows(M, r, x)
            return(True)
        x += 1
        
    # False
    return(False)

def clear(M, r, v):
    # Mrow, col
    row, col = shape(M)
    # , 
    x = r + 1
    while x < row:
        # , 
        if is_zero(M[x][v]):
            x += 1
            continue
        
        # base
        base = - M[x][v] / M[r][v]
        # , M[x] + M[r] * base
        addScaledRow(M, x, r, base)
        x += 1

# RREFM[r][v] == 1
def equal_one(M, r, v):
    # base
    base = 1 / M[r][v]
    # M[r][v]1, M[r][v] * base
    scaleRow(M, r, base)
    return(True)

def clear_above(M, r, v):
    # 
    x = r - 1
    while x >= 0:
        # 0, 
        if is_zero(M[x][v]):
            x -= 1
            continue
            
        # base
        base = - M[x][v]
        # , M[x] + M[r] * base
        addScaledRow(M, x, r, base)
        x -= 1

    # 
    return(True)

def gj_Solve(A, b, decPts = 4, epsilon = 1.0e-16):

    arow, acol = shape(A)
    brow, bcol = shape(b)

        return(None)

    # Mrow, col
    row, col = shape(M)
    # col
    v = 0
    # row
    for r in range(row):
        while v < acol:
            # M[r][v], row, , , 
            if is_zero(M[r][v]):
                if not trans_row(M, r, v):
                    v += 1
                    continue

            # 
            r += 1
            break

        v += 1
