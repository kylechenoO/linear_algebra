def transpose(M):
    # 获取M的row, col数量
    row, col = shape(M)
    
    # Trans Matrix的大小应该与原[row, col]横竖交换, 为[col, row]
    # 初始化trans matrix
    trans = [ [ 0 for c in range(row) ] for r in range(col) ]
    
    # 遍历M, 并未trans matrix赋值
    r = 0
    while r < row:
        c = 0
        while c < col:
            trans[c][r] = M[r][c] 
            c += 1
        r += 1
        
    # 返回Trans Matrix
    return(trans)

def matxMultiply(A, B):
    # 获取A,B的row, col数量
    arow, acol = shape(A)
    brow, bcol = shape(B)
    p = acol
    
    # 两个矩阵相乘, 必须满足A的列数 == B的行数, 否则抛出异常
    if (acol != brow):
        raise ValueError('A.Col must Equal B.Row')
        
    # 遍历A的每一个row
    result = []
    for x in range(arow):
        # 声明一个临时变量用于存储新的result[row]
        row_list = []
        # 遍历B的每一个col
        for y in range(bcol):
            # 获取当前col的list
            col_list = get_col(B, y)
            # 将A[x]列表中的每个元素与B[n][y]中对应的元素相乘并求和, n为变量
            rs = 0
            for i in range(p):
                rs += A[x][i] * col_list[i]

            # 将计算出来的结果存入row_list中
            row_list.append(rs)
        # 将row_list存入result[row]
        result.append(row_list)
    # 返回result
    return(result)

def augmentMatrix(A, b):
    # 获取A,B的row, col数量
    arow, acol = shape(A)
    brow, bcol = shape(b)
    row = arow
    
    # 构造增广矩阵, 必须满足A的行数 == B的行数, 否则抛出异常
    if (arow != brow):
        raise ValueError('A.Row must Equal B.Row')

    result = deepcopy(A)
    # 在result[row]的最后添加b, 合并成增广矩阵
    for x in range(row):
        result[x].append(b[x][0])
    
    # 返回result
    return(result)

2.3 Gaussian Jordan 消元法求解 Ax = b¶

2.3.1 算法¶

步骤1 检查A，b是否行数相同

步骤2 构造增广矩阵Ab

步骤3 逐列转换Ab为化简行阶梯形矩阵 中文维基链接

对于Ab的每一列（最后一列除外）
    当前列为列c
    寻找列c中 对角线以及对角线以下所有元素（行 c~N）的绝对值的最大值
    如果绝对值最大值为0
        那么A为奇异矩阵，返回None (你可以在选做问题2.4中证明为什么这里A一定是奇异矩阵)
    否则
        使用第一个行变换，将绝对值最大值所在行交换到对角线元素所在行（行c） 
        使用第二个行变换，将列c的对角线元素缩放为1
        多次使用第三个行变换，将列c的其他元素消为0

步骤4 返回Ab的最后一列

注： 我们并没有按照常规方法先把矩阵转化为行阶梯形矩阵，再转换为化简行阶梯形矩阵，而是一步到位。如果你熟悉常规方法的话，可以思考一下两者的等价性。

# TODO 实现 Gaussain Jordan 方法求解 Ax = b
""" Gaussian Jordan 方法求解 Ax = b.
    参数
        A: 方阵 
        b: 列向量
        decPts: 四舍五入位数，默认为4
        epsilon: 判读是否为0的阈值，默认 1.0e-16
        
    返回列向量 x 使得 Ax = b 
    返回None，如果 A，b 高度不同
    返回None，如果 A 为奇异矩阵
"""
# 判断是否为零
def is_zero(value, eps = 1.0e-10):
    return(abs(value) < eps)

# 查找第一个不为零的index # error here
def first_nonzero_index(input_list):
    for i, data in enumerate(input_list):
        if not is_zero(data):
            return(i)

# 查找第一个非零的系数
def indices_of_first_nonzero_terms_in_each_row(M):
    # 获取等式个数
    row, col = shape(M)
    num_equations = row

    # 获取系数维度
    num_variables = col - 1

    # 初始化indices列表
    indices = [-1] * num_equations
    # 遍历赋值
    for i, p in enumerate(M):
        try:
            indices[i] = first_nonzero_index(p)
            if indices[i] == (col - 1):
                indices[i] = -1

        except Exception as e:
                raise e
                
    # 返回indices列表
    return(indices)

# 向下查找非零值
def trans_row(M, r, v):
    
    # 获取等式个数
    row, col = shape(M)
    
    # 向下查找M[n][v], n为变量的非零值, 如果有非零值, 则交换到M[r]
    x = r + 1
    while x < row:
        if not is_zero(M[x][v]):
            swapRows(M, r, x)
            return(True)
        x += 1
        
    # 否则返回False
    return(False)

# 向下消元
def clear(M, r, v):
    # 计算M的row, col大小
    row, col = shape(M)
    # 从最后一行开始遍历, 向上消元
    x = r + 1
    while x < row:
        # 如果为零, 则进入下一行消元
        if is_zero(M[x][v]):
            x += 1
            continue
        
        # 设置base值
        base = - M[x][v] / M[r][v]
        # 消元运算, M[x] + M[r] * base
        addScaledRow(M, x, r, base)
        x += 1

# RREF使当前M[r][v] == 1
def equal_one(M, r, v):
    # 设置base
    base = 1 / M[r][v]
    # 使M[r][v]变为1, M[r][v] * base
    scaleRow(M, r, base)
    return(True)

# 向上消元求解
def clear_above(M, r, v):
    # 从当前行的上一行开始遍历
    x = r - 1
    while x >= 0:
        # 如果此元素为0, 则进入下一行操作
        if is_zero(M[x][v]):
            x -= 1
            continue
            
        # 设置base值
        base = - M[x][v]
        # 向上消元求解, M[x] + M[r] * base
        addScaledRow(M, x, r, base)
        x -= 1

    # 返回值
    return(True)

# 高斯消元求解函数
def gj_Solve(A, b, decPts = 4, epsilon = 1.0e-16):    
    # 获取A,B的row, col数量
    arow, acol = shape(A)
    brow, bcol = shape(b)
    
    # 如果A, b高度不同, 则不能构造增广矩阵
    if arow != brow:
        return(None)
    
    # 构造增广矩阵
    M = augmentMatrix(A, b)
    # 获取M的row, col大小
    row, col = shape(M)
    # 从第一个col开始轮询
    v = 0
    # 从第一个row开始轮询
    for r in range(row):
        while v < acol:
            # 如果M[r][v]系数为零, 则查找后面的row, 将非零值交换到此行, 如若没有, 则进入下一个系数的消元计算
            if is_zero(M[r][v]):
                if not trans_row(M, r, v):
                    v += 1
                    continue
            
            # 向下消元
            clear(M, r, v)
            # 进入下一行
            r += 1
            break
        # 进入下一个系数消元计算
        v += 1

    # 获取每个row第一个非零值列表
    indices_list = indices_of_first_nonzero_terms_in_each_row(M)
    
    # 判断奇异矩阵
    for r in range(row):
        if indices_list[r] < 0:
            return(None)
    
    # 设置index的最大值, 从下往上遍历
    r = row - 1
    while r >= 0:
        # 判断是否有有效系数, 如若没有有效系数, 则进入上一个row进行向上消元求解操作
        v = indices_list[r]
        if v < 0:
            r -= 1
            continue
            
        # 除数0异常处理
        if M[r][v] == 0:
            r -= 1
            continue
            
        # 对M[r]求解, 使M[r][v]为1
        equal_one(M, r, v)
        # 向上消元求解
        clear_above(M, r, v)
        # 进入上一行操作
        r -= 1

    # 四舍五入
    matxRound(M)
    
    # 组成结果
    result = [ [M[x][col - 1]] for x in range(row) ]
                
    # 返回值
    return(result)

def calculateMSE(X,Y,m,b):
    # 获取list X, Y的长度
    xsize = len(X)
    ysize = len(Y)
    
    # 如果X, Y长度不等, 则抛出异常, 不能进行后续计算
    if xsize != ysize:
        raise ValueError('Size of X, Y must Equal')
        
    # 设定base值
    base = xsize
    # 设定result变量, 用于存储结果
    result = 0
    # 进行迭代累加
    for i in range(base):
        result += pow((Y[i] - m * X[i] - b), 2)
        
    # 计算平方误差
    result = result / base
    # 返回结果
    return(result)

3.3 (选做) 找到参数 $m, b$ 使得平方平均误差最小¶

这一部分需要简单的微积分知识( $ (x^2)' = 2x $ )。因为这是一个线性代数项目，所以设为选做。

刚刚我们手动调节参数，尝试找到最小的平方平均误差。下面我们要精确得求解 $m, b$ 使得平方平均误差最小。

定义目标函数 $E$ 为 $$ E = \frac{n}{2}\sum_{i=1}^{n}{(y_i - mx_i - b)^2} $$

因为 $E = \frac{n}{2}MSE$, 所以 $E$ 取到最小值时，$MSE$ 也取到最小值。要找到 $E$ 的最小值，即要找到 $m, b$ 使得 $E$ 相对于 $m$, $E$ 相对于 $b$ 的偏导数等于0.

因此我们要解下面的方程组。

$$ \begin{cases} \displaystyle \frac{\partial E}{\partial m} =0 \\ \\ \displaystyle \frac{\partial E}{\partial b} =0 \\ \end{cases} $$
